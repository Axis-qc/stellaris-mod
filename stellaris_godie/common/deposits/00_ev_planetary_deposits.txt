d_ev_meg_freezer = {
	time = 360
	is_for_colonizable = yes
	
	category = deposit_cat_blockers
	resources = {
		category = deposit_blockers
		cost = {
			energy = 1000
		}
	}
	resources = {
		category = planet_deposits
	}
	potential = {
	}
	planet_modifier = {
		planet_max_districts_add = -1
	}
	on_cleared = {
		hidden_effect = {
			random_list = {
				33 = {
					owner = {
						country_event = {
							id = ev_planet_correct.7200
						}
					}
				}
				33 = {
					while = {
						count = 2
						if = {
							limit = {
								any_owned_pop_group = {
									is_gestalt = no
									is_robot_pop_group = no
								}
							}
							random_owned_pop_group = {
								limit = {
									is_gestalt = no
									is_robot_pop_group = no
								}
								root = {
									create_pop_group = {
										size = 100
										species = prev
									}
								}
							}
						}
						else = {
							random_owned_pop_group = {
								root = {
									create_pop_group = {
										size = 100
										species = prev
									}
								}
							}
						}
					}
					owner = {
						country_event = {
							id = ev_planet_correct.7201
						}
					}
				}
				33 = {
					create_country = {
						name = "NAME_mutant_organism"
						type = faction
						effect = {
							establish_communications_no_message = root.owner
							establish_contact = {
								who = root.owner
								location = root.solar_system
							}
						}
					}
					while = {
						count = 4
						create_army = {
							name = "NAME_mutant_organism"
							owner = last_created_country
							species = root.owner_main_species
							type = enraged_army
						}
					}
					owner = {
						country_event = {
							id = ev_planet_correct.7202
						}
					}
				}
			}
		}
	}
	ai_weight = {
		weight = 1
	}
	drop_weight = {
		weight = 0
	}
}
