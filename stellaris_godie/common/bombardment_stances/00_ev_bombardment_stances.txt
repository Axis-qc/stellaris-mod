steal_ev = {
	trigger = {
		owner = {
			NOT = {
				has_policy_flag = orbital_bombardment_steal_no
			}
			OR = {
				is_default_ev_country_type = yes
			}
		}
	}
	
	default = yes
	stop_when_armies_dead = no
	abduct_pops = no
	planet_damage = 0.25
	army_damage = 0.3
	kill_pop_chance = {
		base = 0.12
	}
	min_pops_to_kill_pop = 10
	# root = fleet
	# from = planet
	ai_weight = {
		weight = 9
		modifier = {
			factor = 2
			root.owner = {
				any_war = {
					attacker = {
						is_same_value = root
					}
					using_war_goal = {
						type = wg_bombardment_steal_ev
						owner = root
					}
				}
			} 
			
		}
		modifier = {
			factor = 2.5
			root.owner = {
				OR = {
					has_policy_flag = orbital_bombardment_steal_1th
					has_policy_flag = orbital_bombardment_steal_all
				}
				OR = {
					has_deficit = energy
					has_deficit = minerals
					has_deficit = food
				}
			}
			
		}
		modifier = {
			factor = 2.5
			root.owner = {
				OR = {
					has_policy_flag = orbital_bombardment_steal_2th
					has_policy_flag = orbital_bombardment_steal_all
				}
				OR = {
					has_deficit = alloys
					has_deficit = consumer_goods
				}
			}
		}
		modifier = {
			factor = 1.5
			root.owner = { has_ascension_perk = ap_orbital_bombardment_steal }
		}
		modifier = {
			factor = 1000
			has_country_flag = test_steal_ev
		}
		
	}
}
