country_event = {
	id = ev_resource.9
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		
		if = {
			limit = { 
				any_country = {
					NOT = {
						has_country_flag = ev_void_country
					}
				}
			}
			create_country = {
				name = "NAME_void_country"
				type = the_void
				flag = {
					icon = {
						category = "pirate"
						file = "flag_pirate_6.dds"
					}
					background = {
						category = "backgrounds"
						file = "sinus.dds"
					}
					colors = {
						"black"
						"black"
						"null"
						"null"
					}
				}
				effect = {
					establish_communications_no_message = root
					save_global_event_target_as = ev_void_country
					set_country_flag = ev_void_country
				}
			}
			
			root = {
				country_event = {
					id = ev_resource.10
				}
				set_country_flag = has_void_country
			}
		}
		else = {
			event_target:ev_void_country = {
				establish_communications_no_message = root
			}
			root = {
				country_event = {
					id = ev_resource.10
				}
				set_country_flag = has_void_country
			}
		}
		
	}
}

				which = ev_ev_learn_3_base

					trigger = fleet_power

					which = ev_ev_learn_3_base_fleet
