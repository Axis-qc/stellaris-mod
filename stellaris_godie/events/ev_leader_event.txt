country_event = {
	id = ev_leader.900
	hide_window = yes
	is_triggered_only = yes
	#trigger = {
	#is_default_ev_country_type = yes
	#}
	immediate = {
		#雇佣兵
		if = {
			limit = {
				is_default_ev_country_type = yes
			}
			random_list = {
				33 = {
					from = {
						if = {
							limit = {
								leader_class = commander
							}
							add_trait = leader_trait_mercenary_warrior
						}
						if = {
							limit = {
								leader_class = general
							}
							add_trait = leader_trait_general_mercenary_warrior
						}
					}
				}
				66 = {
				}
			}
		}
		#天选者
		if = {
			limit = {
				has_technology = tech_ultimate_answer
			}
			random_list = {
				20 = {
					from = {
						add_ruler_trait = leader_trait_ruler_chosen
						if = {
							limit = {
								leader_class = official
							}
							add_trait = leader_trait_governor_chosen
						}
						if = {
							limit = {
								leader_class = scientist
							}
							add_trait = leader_trait_scientist_chosen
						}
						if = {
							limit = {
								leader_class = commander
							}
							add_trait = leader_trait_admiral_chosen
						}
						if = {
							limit = {
								leader_class = general
							}
							add_trait = leader_trait_general_chosen
						}
					}
				}
				80 = {
				}
			}
		}
		#科学家等级+3
		if = {
			limit = {
				OR = {
					has_civic = civic_ev_exploration_of_space
				}
				from = {
					leader_class = scientist
				}
			}
			from = {
				add_skill = 3
			}
		}
		#格式塔传统效果
		if = {
			limit = {
				has_tradition = tr_ev_void_5
				is_default_ev_country_type = yes
				OR = {
					is_hive_empire = yes
					is_machine_empire = yes
				}
			}
			from = {
				add_skill = 3
			}
		}
		#特质
		if = {
			limit = {
				from = {
					species = {
						has_trait = trait_ev_skills_add
					}
				}
				#any_owned_pop_group = { has_trait = trait_ev_skills_add }
			}
			from = {
				if = {
					limit = {
						leader_class = official
					}
					add_skill = 1
				}
			}
			#from = {
			#	add_trait = trait_ruler_ev_da_khan
			#}
		}
		#高级特质
		if = {
			limit = {
				#any_owned_pop_group = { has_trait = trait_ev_advanced_research }
				from = {
					species = {
						has_trait = trait_ev_advanced_research
					}
				}
			}
			from = {
				add_skill = 3
			}
		}
		#高级特质
		if = {
			limit = {
				#any_owned_pop_group = { has_trait = trait_ev_advanced_war }
				from = {
					species = {
						has_trait = trait_ev_advanced_war
					}
				}
			}
			from = {
				if = {
					limit = {
						leader_class = commander
					}
					add_skill = 4
				}
			}
		}
		if = {
			limit = {
				from = {
					is_same_value = heir
				}
			}
			from = {
				add_skill = 4
			}
		}
	}
}
